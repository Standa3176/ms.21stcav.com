    ```php
    namespace App\Domain\Agents\Clients;

    use App\Domain\Agents\Enums\FinishReason;

    /**
     * Captures the Prism response shape needed for AgentRun forensics + cost accounting.
     * Maps Prism's enum-named finish reasons to our local FinishReason enum (D-06 schema).
     */
    final readonly class ClaudeResponse
    {
        /**
         * @param array<int, array{name: string, arguments: string|array<mixed>, id?: string}> $toolCalls
         * @param array<int, mixed> $steps
         * @param array<int, mixed> $responseMessages
         */
        public function __construct(
            public string $text,
            public FinishReason $finishReason,
            public int $promptTokens,
            public int $completionTokens,
            public int $costPence,
            public ?string $langfuseTraceId,
            public array $toolCalls,
            public array $steps,
            public array $responseMessages,
        ) {}

        /** Map Prism's FinishReason name string to our enum. */
        public static function mapFinishReason(string $prismName): FinishReason
        {
            return match ($prismName) {
                'Stop' => FinishReason::EndTurn,
                'ToolCalls' => FinishReason::ToolUse,
                'Length' => FinishReason::MaxTokens,
                'StopSequence' => FinishReason::StopSequence,
                'ContentFilter', 'Error', 'Unknown', 'Other' => FinishReason::Error,
                default => FinishReason::Error,
            };
        }
    }
    ```

    ```php
    declare(strict_types=1);

    namespace App\Domain\Agents\Services;

    use RuntimeException;

    final class CostCalculator
    {
        public function compute(int $promptTokens, int $completionTokens, string $model): int
        {
            $rates = config("agents.pricing.{$model}");
            if (! is_array($rates) || ! isset($rates['input_pence_per_token'], $rates['output_pence_per_token'])) {
                throw new RuntimeException("No pricing configured for model: {$model}");
            }
            return (int) ceil(
                ($promptTokens * (float) $rates['input_pence_per_token'])
                + ($completionTokens * (float) $rates['output_pence_per_token'])
            );
        }
    }
    ```

    **Create `app/Domain/Agents/Clients/ClaudeClient.php`** — single Prism chokepoint:
    ```php
    declare(strict_types=1);

    namespace App\Domain\Agents\Clients;

    use App\Domain\Agents\Services\CostCalculator;
    use App\Foundation\Integration\Services\IntegrationLogger;
    use Illuminate\Support\Facades\Context;
    use Prism\Prism\Enums\Provider;
    use Prism\Prism\Prism;

    /**
     * Sole wrapper around prism-php/prism for Anthropic calls. Every agent's LLM
     * traffic flows through here so IntegrationLogger captures the HTTP shape
     * and BudgetGuard records actuals post-flight (Plan 03 wires the BudgetGuard call).
     *
     * AGNT-07: model=claude-sonnet-4-6, temperature=0, withMaxSteps(8), withMaxTokens(4000)
     */
    final class ClaudeClient
    {
        public const DEFAULT_MODEL = 'claude-sonnet-4-6';
        public const DEFAULT_MAX_STEPS = 8;
        public const DEFAULT_MAX_TOKENS = 4000;
        public const DEFAULT_TEMPERATURE = 0.0;

        public function __construct(
            private readonly CostCalculator $costCalculator,
            private readonly IntegrationLogger $logger,
        ) {}

        /**
         * @param array<int, \Prism\Prism\ValueObjects\Messages\UserMessage|\Prism\Prism\ValueObjects\Messages\AssistantMessage|\Prism\Prism\ValueObjects\Messages\ToolResultMessage> $messages
         * @param array<int, \Prism\Prism\Tool\Tool> $tools
         */
        public function generate(
            string $systemPrompt,
            array $messages,
            array $tools = [],
            ?string $model = null,
            int $maxSteps = self::DEFAULT_MAX_STEPS,
            int $maxTokens = self::DEFAULT_MAX_TOKENS,
            float $temperature = self::DEFAULT_TEMPERATURE,
        ): ClaudeResponse {
            $model ??= (string) config('agents.default_model', self::DEFAULT_MODEL);

            $request = Prism::text()
                ->using(Provider::Anthropic, $model)
                ->withSystemPrompt($systemPrompt)
                ->withMessages($messages)
                ->withMaxSteps($maxSteps)
                ->withMaxTokens($maxTokens)
                ->usingTemperature($temperature)
                ->withClientRetry(2, 100);

            if (! empty($tools)) {
                $request = $request->withTools($tools);
            }

            $startedAt = microtime(true);
            $response = $request->asText();
            $latencyMs = (int) round((microtime(true) - $startedAt) * 1000);

            $promptTokens = (int) ($response->usage->promptTokens ?? 0);
            $completionTokens = (int) ($response->usage->completionTokens ?? 0);
            $costPence = $this->costCalculator->compute($promptTokens, $completionTokens, $model);

            $this->logger->record(
                provider: 'anthropic',
                endpoint: '/v1/messages',
                method: 'POST',
                statusCode: 200,
                latencyMs: $latencyMs,
                metadata: [
                    'model' => $model,
                    'prompt_tokens' => $promptTokens,
                    'completion_tokens' => $completionTokens,
                    'cost_pence' => $costPence,
                    'finish_reason' => $response->finishReason->name,
                ],
            );

            return new ClaudeResponse(
                text: (string) ($response->text ?? ''),
                finishReason: ClaudeResponse::mapFinishReason($response->finishReason->name),
                promptTokens: $promptTokens,
                completionTokens: $completionTokens,
                costPence: $costPence,
                langfuseTraceId: $this->extractLangfuseTraceId(),
                toolCalls: $response->toolCalls ?? [],
                steps: $response->steps ?? [],
                responseMessages: $response->responseMessages ?? [],
            );
        }

        /**
         * mliviu79/laravel-langfuse-prism pushes trace_id into the Laravel Context after each call.
         * If the shim isn't present (test mode, custom-otel fallback), returns null.
         *
         * Q2 RETIREMENT (per RESEARCH §Open Questions Q2):
         *   Plan 02 Task 2 Test 10 verifies this Context::get path works against Prism::fake().
         *   Fallback path if the shim does NOT populate Context: read X-Langfuse-Trace-Id from
         *   the Prism HTTP response via a Prism HTTP middleware. Documented in observability.md.
         */
        private function extractLangfuseTraceId(): ?string
        {
            $traceId = Context::get('langfuse_trace_id');
            return is_string($traceId) ? $traceId : null;
        }
    }
    ```

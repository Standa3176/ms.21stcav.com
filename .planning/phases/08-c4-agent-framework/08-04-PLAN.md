    declare(strict_types=1);

    namespace App\Domain\Agents\Services\Tools;

    use Prism\Prism\Facades\Tool as PrismTool;

    final class ReadHealthCheckTool extends Tool
    {
        public function name(): string { return 'read_health_check'; }

        public function description(): string
        {
            return 'Returns the current timestamp, git SHA, and app version. Call once to confirm framework health. Do NOT retry.';
        }

        public function asPrismTool(): \Prism\Prism\Tool\Tool
        {
            return PrismTool::as($this->name())
                ->for($this->description())
                ->using(fn (): string => json_encode([
                    'timestamp' => now()->toIso8601String(),
                    'git_sha' => $this->resolveGitSha(),
                    'app_version' => (string) config('app.version', 'v2.0'),
                ], JSON_THROW_ON_ERROR));
        }

        private function resolveGitSha(): string
        {
            $sha = trim((string) @shell_exec('git rev-parse HEAD 2>/dev/null'));
            return $sha !== '' ? substr($sha, 0, 12) : 'unknown';
        }
    }

    namespace App\Domain\Agents\Agents;

    use App\Domain\Agents\Contracts\RunsAsAgent;
    use App\Domain\Agents\Enums\TrustTier;
    use App\Domain\Agents\Guardrails\OutboundRegexFilterGuardrail;
    use App\Domain\Agents\Guardrails\SensitiveFieldsStripGuardrail;
    use App\Domain\Agents\Services\PromptRenderer;
    use App\Domain\Agents\Services\Tools\ReadHealthCheckTool;
    use App\Domain\Agents\ValueObjects\AgentResult;

    /**
     * Smoke-test agent. NOT a business consumer — proves the framework end-to-end
     * (registry → BudgetGuard → ToolBus → ClaudeClient → Langfuse → SuggestionApplierResolver
     * → Filament). EchoAgent stays in the repo as the canonical "framework health" pattern
     * even after Phase 10's PricingAgent ships (per CONTEXT §specifics).
     */
    final class EchoAgent implements RunsAsAgent
    {
        public function __construct(
            private readonly PromptRenderer $promptRenderer,
        ) {}

        public static function kind(): string { return 'echo'; }
        public static function trustTier(): TrustTier { return TrustTier::Trusted; }

        public function tools(): array
        {
            return [app(ReadHealthCheckTool::class)];
        }

        public function systemPrompt(array $context = []): string
        {
            return $this->promptRenderer->render('echo', $context)['prompt'];
        }

        public function guardrails(): array
        {
            // Trusted tier — skip PromptInjectionXmlFence (no customer input).
            // Outbound regex filter still applies as a defence-in-depth measure.
            return [
                app(SensitiveFieldsStripGuardrail::class),
                app(OutboundRegexFilterGuardrail::class),
            ];
        }

        public function execute(array $input, TrustTier $tier): AgentResult
        {
            // EchoAgent's execute() is a stub — RunAgentJob calls it but the actual
            // ClaudeClient call + result wiring happens in RunAgentJob, which then
            // builds the AgentResult. This method exists to satisfy the contract;
            // RunAgentJob does NOT invoke it for the echo flow because RunAgentJob
            // owns the orchestration. (See RunAgentJob::handle() — it uses the
            // contract's getter methods directly, not execute().)
            //
            // Future agents may override this if they need pre/post processing
            // beyond what RunAgentJob orchestrates.
            throw new \LogicException(
                'EchoAgent::execute is a stub — RunAgentJob owns the orchestration.'
            );
        }
    }
    ```

    NOTE: This codifies the architectural choice from RESEARCH §Pattern 2 — `RunAgentJob::handle()` is the single orchestration point, NOT `RunsAsAgent::execute()`. The contract's `execute()` exists as a forward-compatibility seam for future agents that need to wrap the orchestration (none in v2.0). Plan 03's RunsAsAgent contract retains the `execute` method for that future flexibility, but Plan 04's RunAgentJob calls `tools()`/`systemPrompt()`/`guardrails()` directly. Document this clearly in 08-04-SUMMARY.md.

    use Prism\Prism\ValueObjects\Messages\UserMessage;

    final class RunAgentJob implements ShouldQueue
    {
        use Queueable, InteractsWithQueue, SerializesModels;

        public string $queue = 'agents';
        public int $tries = 1;
        public int $timeout = 180;

        public function __construct(
            public readonly string $kind,
            public readonly array $input = [],
            public readonly ?string $triggeringSuggestionId = null,
            public readonly ?string $triggeringCorrelationId = null,
        ) {}

        public function handle(
            AgentRegistry $registry,
            BudgetGuard $budgetGuard,
            ToolBus $toolBus,
            GuardrailEngine $guardrailEngine,
            ClaudeClient $client,
            AgentSuggestionWriter $writer,
            PromptRenderer $promptRenderer,
        ): void {
            $correlationId = $this->triggeringCorrelationId
                ?? (string) Context::get('correlation_id')
                ?? (string) Str::uuid();

            $agent = $registry->resolve($this->kind);
            $rendered = $promptRenderer->render($this->kind, $this->input);

            $run = AgentRun::create([
                'kind' => $this->kind,
                'status' => AgentRunStatus::Running->value,
                'triggering_suggestion_id' => $this->triggeringSuggestionId,
                'triggering_correlation_id' => $correlationId,
                'system_prompt_hash' => $rendered['hash'],
                'tool_calls' => [],
                'started_at' => now(),
            ]);

            event(new AgentRunStarted($run));

            // Track which phase we're in so the GuardrailViolationException catch
            // can record `when: 'pre' | 'post'` on agent_runs.guardrail_failures
            // (per plan-checker iter 1 BLOCKER 1).
            $guardrailPhase = 'pre';

            try {
                // Pre-flight budget gate (D-01..D-05)
                $budgetGuard->assertHasBudget($this->kind);

                // Pre-flight guardrails (PromptInjectionXmlFence on Untrusted etc.)
                $tier = $agent::trustTier();
                $sanitisedInput = $guardrailEngine->runPreFlight($agent, $this->input, $tier);

                // Render messages (single user message for v2.0; multi-turn deferred)
                $userText = trim(json_encode($sanitisedInput, JSON_THROW_ON_ERROR));
                $messages = [new UserMessage($userText)];

                $prismTools = $toolBus->buildPrismTools($agent->tools(), $run);

                // The actual Anthropic call (or Prism::fake() in tests)
                $response = $client->generate(
                    systemPrompt: $rendered['prompt'],
                    messages: $messages,
                    tools: $prismTools,
                );

                // Post-flight guardrails (OutboundRegexFilter, etc.)
                $guardrailPhase = 'post';
                $response = $guardrailEngine->runPostFlight($agent, $response, $tier);

                // Build tool_calls JSON from response steps (4KB cap per entry)
                $toolCallsLog = $this->extractToolCallsFromSteps($response->steps, $toolBus);

                // Persist run completion BEFORE writing suggestion (so AgentRun row is final
                // truth even if downstream applier registration is missing).
                $run->update([
                    'status' => AgentRunStatus::Completed->value,
                    'completed_at' => now(),
                    'finish_reason' => $response->finishReason->value,
                    'tool_calls' => $toolCallsLog,
                    'agent_reasoning_summary' => mb_substr($response->text, 0, 8192),
                    'prompt_token_count' => $response->promptTokens,
                    'completion_token_count' => $response->completionTokens,
                    'cost_pence' => $response->costPence,
                    'langfuse_trace_id' => $response->langfuseTraceId,
                ]);

                // Record actual spend (post-flight per CONTEXT D-01)
                $budgetGuard->recordSpend($this->kind, $response->costPence);

                // For EchoAgent: produce one shadow Suggestion of kind=echo_health with
                // health-check evidence pulled from the tool result.
                $draft = new SuggestionDraft(
                    kind: 'echo_health',
                    payload: ['summary' => $response->text],
                    evidence: [
                        'agent_run_id' => $run->id,
                        'tool_calls' => $toolCallsLog,
                        'cost_pence' => $response->costPence,
                    ],
                );
                $writer->write($draft, $run);

                event(new AgentRunCompleted($run));
            } catch (MonthlyBudgetExceededException $e) {
                $run->update([
                    'status' => AgentRunStatus::MonthlyBudgetBlocked->value,
                    'completed_at' => now(),
                    'agent_reasoning_summary' => $e->getMessage(),
                ]);
                event(new AgentRunFailed($run, $e));
                throw $e;
            } catch (BudgetExceededException $e) {
                $run->update([
                    'status' => AgentRunStatus::BudgetExceeded->value,
                    'completed_at' => now(),
                    'agent_reasoning_summary' => $e->getMessage(),
                ]);
                event(new AgentRunFailed($run, $e));
                throw $e;
            } catch (GuardrailViolationException $e) {
                // Plan-checker iter 1 BLOCKER 1 — capture violation reason on
                // agent_runs.guardrail_failures (15th column shipped by Plan 01)
                $run->update([
                    'status' => AgentRunStatus::GuardrailBlocked->value,
                    'completed_at' => now(),
                    'guardrail_failures' => [[
                        'guardrail' => $e->guardrailClass !== '' ? $e->guardrailClass : GuardrailViolationException::class,
                        'message' => $e->getMessage(),
                        'when' => $guardrailPhase,
                        'occurred_at' => now()->toIso8601String(),
                    ]],
                    'agent_reasoning_summary' => mb_substr($e->getMessage(), 0, 8192),
                ]);
                // Write a guardrail_blocked suggestion (kind=agent_guardrail_blocked) per AGNT-06
                $writer->write(new SuggestionDraft(
                    kind: 'agent_guardrail_blocked',
                    payload: ['violation' => $e->getMessage()],
                    evidence: ['run_id' => $run->id, 'guardrail' => $e->guardrailClass, 'when' => $guardrailPhase],
                ), $run);
                event(new AgentRunFailed($run, $e));
                throw $e;
            } catch (\Throwable $e) {
                $run->update([
                    'status' => AgentRunStatus::Failed->value,
                    'completed_at' => now(),
                    'agent_reasoning_summary' => $e->getMessage(),
                ]);
                event(new AgentRunFailed($run, $e));
                throw $e;
            }
        }

        /** Walk Prism's response->steps and extract tool calls + results for AgentRun.tool_calls. */
        private function extractToolCallsFromSteps(array $steps, ToolBus $toolBus): array
        {
            $log = [];
            foreach ($steps as $step) {
                // Each step may contain toolCalls + toolResults arrays. Adapt to Prism's
                // actual step shape — we read both arrays defensively.
                $toolCalls = (is_object($step) && property_exists($step, 'toolCalls')) ? $step->toolCalls : [];
                $toolResults = (is_object($step) && property_exists($step, 'toolResults')) ? $step->toolResults : [];
                foreach ($toolCalls as $i => $call) {
                    $name = is_object($call) ? ($call->name ?? 'unknown') : ($call['name'] ?? 'unknown');
                    $args = is_object($call) ? ($call->arguments ?? '{}') : ($call['arguments'] ?? '{}');
                    $result = $toolResults[$i] ?? null;
                    $resultValue = is_object($result) ? ($result->result ?? null) : (is_array($result) ? ($result['result'] ?? null) : null);
                    $log[] = [
                        'tool_name' => $name,
                        'inputs' => $toolBus->truncate($args, 4096),
                        'outputs' => $toolBus->truncate($resultValue ?? '', 4096),
                        'tokens_used' => 0,  // Prism may not break out per-tool tokens; aggregate sits on AgentRun
                        'latency_ms' => 0,
                    ];
                }
            }
            return $log;
        }
    }
    ```

                                {kind : Agent kind to dispatch (e.g. echo, pricing)}
                                {--dry-run : Run synchronously instead of dispatching to the agents queue}
                                {--input= : Optional JSON-encoded input payload}';

        protected $description = 'Dispatch an agent run; --dry-run executes synchronously for verification';

        protected function perform(): int
        {
            $kind = (string) $this->argument('kind');
            $inputJson = (string) ($this->option('input') ?? '{}');
            $input = json_decode($inputJson, true) ?: [];

            $job = new RunAgentJob(
                kind: $kind,
                input: $input,
                triggeringCorrelationId: (string) Context::get('correlation_id'),
            );

            try {
                if ($this->option('dry-run')) {
                    $this->info("Running {$kind} synchronously (--dry-run)...");
                    dispatch_sync($job);
                } else {
                    $this->info("Dispatching {$kind} to agents queue...");
                    dispatch($job);
                }
                $this->info('OK');
                return 0;
            } catch (\Throwable $e) {
                $this->error('Agent run failed: ' . $e->getMessage());
                return 1;
            }
        }
    }

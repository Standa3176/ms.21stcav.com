    declare(strict_types=1);

    namespace App\Domain\Agents\Console\Commands;

    use App\Console\Commands\BaseCommand;

    /**
     * AGNT-11 ã wraps shield:generate --all --force with automatic P5-F restoration.
     *
     * Workflow:
     *   1. Refuse if git working tree is dirty (unless --force)
     *   2. Capture current policy paths via `git ls-files app/Domain/*\/Policies/*.php`
     *   3. Run `shield:generate --all --force`
     *   4. For each captured path NOT in --allow-new, `git checkout -- {path}`
     *   5. Run PolicyTemplateIntegrityTest as smoke test; exit 1 if it fails
     *
     * --allow-new flag: lets a brand-new policy (e.g. AgentRunPolicy on first run) survive
     * Shield's auto-generation. Subsequent runs commit the hand-written body to git, then
     * --allow-new is no longer needed.
     */
    final class ShieldSafeRegenerateCommand extends BaseCommand
    {
        protected $signature = 'shield:safe-regenerate
                                {--allow-new=* : Policy class names allowed to be newly created without restore}
                                {--restore=true : Restore hand-written policies after regen (default true)}
                                {--force : Proceed even if git working tree is dirty}';

        protected $description = 'AGNT-11 ã Shield wrapper with automatic P5-F policy restoration';

        protected function perform(): int
        {
            // 1. Dirty-tree guard
            $force = (bool) $this->option('force');
            if (! $force && $this->isGitTreeDirty()) {
                $this->error('git working tree is dirty ã commit or stash before running shield:safe-regenerate. Pass --force to override.');
                return 1;
            }

            // 2. Capture policy paths
            $this->info('Step 1: Capturing current policy state from git...');
            $policies = $this->capturePoliciesFromGit();
            $this->info('  Captured ' . count($policies) . ' policy file(s)');

            // 3. Run shield:generate
            $this->info('Step 2: Running shield:generate --all --force...');
            $exitCode = $this->call('shield:generate', ['--all' => true, '--force' => true]);
            if ($exitCode !== 0) {
                $this->error("shield:generate failed with exit code {$exitCode}");
                return $exitCode;
            }

            // 4. Restore policies (unless --restore=false)
            $restore = filter_var($this->option('restore'), FILTER_VALIDATE_BOOLEAN, FILTER_NULL_ON_FAILURE);
            $restore = $restore !== null ? $restore : true;
            if ($restore) {
                $allowNew = (array) $this->option('allow-new');
                $this->info('Step 3: Restoring hand-written policies (P5-F)' . ($allowNew ? ' (skipping ' . implode(', ', $allowNew) . ')' : ''));
                foreach ($policies as $absolutePath) {
                    $className = pathinfo($absolutePath, PATHINFO_FILENAME);
                    if (in_array($className, $allowNew, true)) {
                        $this->info("  Skipping {$className} (--allow-new)");
                        continue;
                    }
                    $rc = 0;
                    $output = [];
                    exec('git checkout -- ' . escapeshellarg($absolutePath) . ' 2>&1', $output, $rc);
                    if ($rc !== 0) {
                        $this->warn("  Could not restore {$className}: " . implode(' ', $output));
                    }
                }
            } else {
                $this->info('Step 3: Skipping restoration (--restore=false)');
            }

            // 5. PolicyTemplateIntegrityTest gate
            $this->info('Step 4: Running PolicyTemplateIntegrityTest...');
            $testExit = $this->call('test', ['--filter' => 'PolicyTemplateIntegrityTest']);
            if ($testExit !== 0) {
                $this->error('PolicyTemplateIntegrityTest FAILED ã Shield {{ Placeholder }} leak detected.');
                return 1;
            }

            $this->info('shield:safe-regenerate complete.');
            return 0;
        }

        private function isGitTreeDirty(): bool
        {
            $output = [];
            $rc = 0;
            exec('git status --porcelain 2>&1', $output, $rc);
            return $rc === 0 && count(array_filter($output, fn ($l) => trim($l) !== '')) > 0;
        }

        /** @return array<string> absolute paths */
        private function capturePoliciesFromGit(): array
        {
            $output = [];
            $rc = 0;
            exec('git ls-files app/Domain/*/Policies/*.php 2>&1', $output, $rc);
            if ($rc !== 0) return [];
            return array_map(fn ($p) => base_path(trim($p)), array_filter($output, fn ($l) => trim($l) !== ''));
        }
    }

    declare(strict_types=1);

    namespace App\Domain\Agents\Services;

    use App\Domain\Agents\Models\AgentRun;
    use Illuminate\Support\Facades\DB;
    use Illuminate\Support\Facades\Log;

    /**
     * D-09 ã GDPR scrub-in-place for AgentRun rows containing customer PII.
     *
     * Mirrors Phase 4 D-13 (gdpr:erase-bitrix-customer scrubs Contact + Company)
     * + Phase 6 GDPR pattern (scrub-in-place; preserve operational integrity).
     *
     * Preserved fields (operational + financial integrity):
     *   - cost_pence, prompt_token_count, completion_token_count
     *   - agent_kind, system_prompt_hash, langfuse_trace_id
     *   - started_at, completed_at, status
     *
     * Scrubbed fields:
     *   - tool_calls JSON entries with customer_email/customer_phone/customer_name
     *     in inputs OR outputs ã values replaced with REDACTED-{sha256-prefix}
     *   - agent_reasoning_summary ã "[scrubbed per GDPR erasure {gdpr_log_ulid}]"
     */
    final class AgentRunGdprScrubber
    {
        public const PII_KEYS = ['customer_email', 'customer_phone', 'customer_name', 'email', 'phone'];

        /**
         * Scrub all AgentRun rows where tool_calls or agent_reasoning_summary contain
         * the given customer email (matched as substring in JSON-encoded form).
         *
         * @return array<string>  list of scrubbed AgentRun IDs
         */
        public function scrubForCustomer(string $customerEmail, string $gdprLogUlid): array
        {
            $hashPrefix = substr(hash('sha256', strtolower($customerEmail)), 0, 12);

            // MySQL JSON_SEARCH path ã MySQL 8.0+ feature (v1 baseline confirms)
            $candidates = AgentRun::query()
                ->whereRaw('JSON_SEARCH(tool_calls, ?, ?) IS NOT NULL', ['one', $customerEmail])
                ->orWhere('agent_reasoning_summary', 'like', '%' . $customerEmail . '%')
                ->get();

            $scrubbedIds = [];
            foreach ($candidates as $run) {
                $this->scrubRun($run, $customerEmail, $hashPrefix, $gdprLogUlid);
                $scrubbedIds[] = $run->id;
            }

            // Write to gdpr_erasure_log (Phase 6 table) listing all scrubbed run IDs
            if (! empty($scrubbedIds)) {
                DB::table('gdpr_erasure_log')->insert([
                    'gdpr_log_ulid' => $gdprLogUlid,
                    'customer_email_hash' => hash('sha256', strtolower($customerEmail)),
                    'agent_run_ids' => json_encode($scrubbedIds),
                    'scrubbed_at' => now(),
                    'created_at' => now(),
                    'updated_at' => now(),
                ]);
            }

            Log::info('AgentRunGdprScrubber: scrubbed runs for GDPR erasure', [
                'gdpr_log_ulid' => $gdprLogUlid,
                'customer_email_hash' => $hashPrefix,
                'agent_run_count' => count($scrubbedIds),
            ]);

            return $scrubbedIds;
        }

        {

            ]);

        private function scrubArray(array $data, string $customerEmail, string $token): array
        {
            return collect($data)->map(function ($value, $key) use ($customerEmail, $token) {
                if (is_array($value)) {
                    return $this->scrubArray($value, $customerEmail, $token);
                }
                if (is_string($value)) {
                    if (in_array(strtolower((string) $key), self::PII_KEYS, true) && stripos($value, $customerEmail) !== false) {
                        return $token;
                    }
                    return str_ireplace($customerEmail, $token, $value);
                }
                return $value;
            })->all();
        }

    {
        protected $signature = 'agents:prune-archive
                                {--days=1825 : Retention horizon in days (default 5 years per D-07)}
                                {--dry-run : Log intent only; do not write archive or delete rows}';

        {
            $days = (int) $this->option('days');
            $dryRun = (bool) $this->option('dry-run');
            $threshold = Carbon::now()->subDays($days);

            $query = AgentRun::query()->where('completed_at', '<', $threshold);

                return 0;

            // Stream rows in chunks to avoid memory blow-up (target 1k rows per chunk)
            $payload = [];
            $query->chunkById(1000, function ($chunk) use (&$payload) {
                foreach ($chunk as $run) {
                    $payload[] = $run->toArray();
                }
            });

            $json = json_encode($payload, JSON_UNESCAPED_SLASHES | JSON_THROW_ON_ERROR);
            $gzipped = gzencode($json, 9);
            Storage::disk('local')->put($archivePath, $gzipped);
            $this->info('Archive written: ' . Storage::disk('local')->size($archivePath) . ' bytes');

            // DELETE after archive write succeeds
            $deleted = AgentRun::query()->where('completed_at', '<', $threshold)->delete();
            $this->info("Deleted {$deleted} row(s)");

            // Audit log entry per D-07

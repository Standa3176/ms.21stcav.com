    ```php
    <?php

    declare(strict_types=1);

    namespace App\Domain\Agents\Tools\Pricing;

    use App\Domain\Agents\Services\Tools\Tool;

    /**
     * 3 KB soft cap helper for PricingAgent tools (CONTEXT D-05).
     *
     * Each subclass implements reduceLargestArray() to define how its payload
     * shrinks — margin_history downsamples evenly; competitor_prices keeps the
     * most-recent N per competitor; supplier_price_trend trims the oldest entries.
     *
     * The cap is a soft signal — if a tool's reduced payload still exceeds
     * SOFT_CAP_BYTES, it returns the reduced payload anyway (better truncated
     * data than no data). _truncated:true + _total_available signals to the
     * agent that data exists beyond the cap.
     */
    abstract class TruncatingTool extends Tool
    {
        protected const SOFT_CAP_BYTES = 3072;

        /**
         * Encode payload + apply soft cap. Adds _truncated/_total_available hints
         * when reduction was needed.
         *
         * @param  array<string, mixed>  $payload
         */
        protected function capJson(array $payload, int $totalAvailable): string
        {
            $json = json_encode($payload, JSON_THROW_ON_ERROR);
            if (strlen($json) <= self::SOFT_CAP_BYTES) {
                return $json;
            }

            $reduced = $this->reduceLargestArray($payload, self::SOFT_CAP_BYTES);
            $reduced['_truncated'] = true;
            $reduced['_total_available'] = $totalAvailable;

            return json_encode($reduced, JSON_THROW_ON_ERROR);
        }

        /**
         * @param  array<string, mixed>  $payload
         * @return array<string, mixed>
         */
        abstract protected function reduceLargestArray(array $payload, int $maxBytes): array;
    }
    ```

    ```php
    final class ReadMarginHistoryTool extends TruncatingTool
    {
        public function name(): string { return 'read_margin_history'; }

        public function description(): string {
            return 'Read the last 90 days of margin changes for a SKU. Returns up to 30 entries (downsampled if more exist) with date, rule scope, old/new bps, delta. Use once per SKU to understand price trajectory.';
        }

        public function asPrismTool(): \Prism\Prism\Tool {
            return \Prism\Prism\Tool::as($this->name())
                ->for($this->description())
                ->withStringParameter('sku', 'The SKU to look up')
                ->using(fn (string $sku): string => $this->execute($sku));
        }

        private function execute(string $sku): string
        {
            $since = now()->subDays(90);

            // Primary: spatie activity_log entries on PricingRule with old margin_basis_points
            $logRows = \Spatie\Activitylog\Models\Activity::query()
                ->where('subject_type', \App\Domain\Pricing\Models\PricingRule::class)
                ->where('created_at', '>=', $since)
                ->whereJsonContainsKey('properties->old.margin_basis_points')
                // Filter to rules whose scope mentions this SKU
                ->get()
                ->filter(fn ($row) => $this->ruleAffectsSku($row->subject_id, $sku))
                ->values();

            // Fallback context: margin_change Suggestion rows referencing this SKU in evidence
            $suggestionRows = \App\Domain\Suggestions\Models\Suggestion::query()
                ->where('kind', 'margin_change')
                ->where('proposed_at', '>=', $since)
                ->whereJsonContains('evidence->sku', $sku)
                ->get();

            $changes = $logRows->map(fn ($row) => [
                'date' => $row->created_at?->toDateString(),
                'rule_scope' => $this->lookupRuleScope($row->subject_id) ?? 'unknown',
                'old_margin_bps' => (int) data_get($row->properties, 'old.margin_basis_points'),
                'new_margin_bps' => (int) data_get($row->properties, 'attributes.margin_basis_points'),
                'delta_bps' => (int) data_get($row->properties, 'attributes.margin_basis_points')
                    - (int) data_get($row->properties, 'old.margin_basis_points'),
                'applied' => true,
                'via' => 'activity_log',
            ])->concat($suggestionRows->map(fn ($s) => [
                'date' => $s->proposed_at?->toDateString(),
                'rule_scope' => (string) data_get($s->evidence, 'pricing_rule.scope', 'unknown'),
                'old_margin_bps' => (int) data_get($s->evidence, 'our_current_margin_bps', 0),
                'new_margin_bps' => (int) data_get($s->evidence, 'proposed_margin_bps', 0),
                'delta_bps' => (int) data_get($s->evidence, 'margin_delta_bps', 0),
                'applied' => $s->status === \App\Domain\Suggestions\Models\Suggestion::STATUS_APPROVED,
                'via' => 'margin_change_suggestion',
            ]))->sortByDesc('date')->values();

            $total = $changes->count();
            // Downsample to 30 entries if needed (preserve most-recent 5 + first 5)
            $capped = $total > 30 ? $this->downsampleEvenly($changes->all(), 30) : $changes->all();

            return $this->capJson([
                'sku' => $sku,
                'window_days' => 90,
                'changes' => $capped,
                '_truncated' => $total > 30,
                '_total_available' => $total,
            ], $total);
        }

        protected function reduceLargestArray(array $payload, int $maxBytes): array
        {
            $payload['changes'] = array_slice($payload['changes'] ?? [], 0, 15);
            return $payload;
        }

        // Helpers — ruleAffectsSku, lookupRuleScope, downsampleEvenly — defined as private methods
    }
    ```

    ```php
    final class ReadCompetitorPricesTool extends TruncatingTool {
        public function name(): string { return 'read_competitor_prices'; }
        public function description(): string { return 'Read the last 90 days of competitor prices for a SKU, grouped by competitor. Returns up to 50 most-recent rows distributed across competitors. Use once per SKU.'; }

        public function asPrismTool(): \Prism\Prism\Tool {
            return \Prism\Prism\Tool::as($this->name())
                ->for($this->description())
                ->withStringParameter('sku', 'The SKU to look up')
                ->using(fn (string $sku): string => $this->execute($sku));
        }

        private function execute(string $sku): string {
            $since = now()->subDays(90);
            $rows = \App\Domain\Competitor\Models\CompetitorPrice::query()
                ->where('sku', $sku)
                ->where('recorded_at', '>=', $since)
                ->with('competitor:id,name')
                ->orderByDesc('recorded_at')
                ->limit(50)
                ->get();

            $totalAvailable = \App\Domain\Competitor\Models\CompetitorPrice::query()
                ->where('sku', $sku)
                ->where('recorded_at', '>=', $since)
                ->count();

            $grouped = $rows->groupBy('competitor_id')->map(fn ($group, $cid) => [
                'competitor_id' => (int) $cid,
                'competitor_name' => $group->first()->competitor?->name ?? "Competitor #{$cid}",
                'data_points' => $group->map(fn ($p) => [
                    'recorded_at' => $p->recorded_at?->toIso8601String(),
                    'price_pennies_ex_vat' => (int) $p->price_pennies_ex_vat,
                ])->values()->all(),
            ])->values();

            return $this->capJson([
                'sku' => $sku,
                'window_days' => 90,
                'competitors' => $grouped->all(),
                '_truncated' => $totalAvailable > $rows->count(),
                '_total_available' => $totalAvailable,
            ], $totalAvailable);
        }

        protected function reduceLargestArray(array $payload, int $maxBytes): array {
            // Drop oldest data_points proportionally per competitor until under cap
            foreach ($payload['competitors'] as &$c) {
                $c['data_points'] = array_slice($c['data_points'], 0, 5);
            }
            return $payload;
        }
    }
    ```

    ```php
    final class ReadSupplierPriceTrendTool extends TruncatingTool {
        public function name(): string { return 'read_supplier_price_trend'; }
        public function description(): string { return 'Read the last 90 days of supplier price changes for a SKU. May return empty data_points if supplier price history is not retained per-product; check current_buy_price_pennies as fallback.'; }

        public function asPrismTool(): \Prism\Prism\Tool {
            return \Prism\Prism\Tool::as($this->name())
                ->for($this->description())
                ->withStringParameter('sku', 'The SKU to look up')
                ->using(fn (string $sku): string => $this->execute($sku));
        }

        private function execute(string $sku): string {
            $product = \App\Domain\Products\Models\Product::query()
                ->where('sku', $sku)
                ->select(['id', 'sku', 'buy_price_pennies'])
                ->first();
            if ($product === null) {
                return json_encode(['sku' => $sku, 'data_points' => [], 'current_buy_price_pennies' => 0, '_note' => 'product not found'], JSON_THROW_ON_ERROR);
            }

            $since = now()->subDays(90);
            $logRows = \Spatie\Activitylog\Models\Activity::query()
                ->where('subject_type', \App\Domain\Products\Models\Product::class)
                ->where('subject_id', $product->id)
                ->where('created_at', '>=', $since)
                ->whereJsonContainsKey('properties->old.buy_price_pennies')
                ->orderByDesc('created_at')
                ->limit(30)
                ->get();

            $points = $logRows->map(fn ($row) => [
                'date' => $row->created_at?->toDateString(),
                'buy_price_pennies' => (int) data_get($row->properties, 'attributes.buy_price_pennies'),
                'old_buy_price_pennies' => (int) data_get($row->properties, 'old.buy_price_pennies'),
            ])->all();

            $payload = [
                'sku' => $sku,
                'window_days' => 90,
                'data_points' => $points,
                'current_buy_price_pennies' => (int) $product->buy_price_pennies,
            ];
            if (empty($points)) {
                $payload['_note'] = 'supplier price history not retained — see current_buy_price_pennies for latest snapshot';
            }

            return $this->capJson($payload, count($points));
        }

        protected function reduceLargestArray(array $payload, int $maxBytes): array {
            $payload['data_points'] = array_slice($payload['data_points'] ?? [], 0, 10);
            return $payload;
        }
    }
    ```

    ```php
    final class ReadSalesVolume90dTool extends TruncatingTool {
        public function name(): string { return 'read_sales_volume_90d'; }
        public function description(): string { return 'Read the cached 90-day sales count for a SKU. Returns sales_count + cache freshness hint.'; }

        public function asPrismTool(): \Prism\Prism\Tool {
            return \Prism\Prism\Tool::as($this->name())
                ->for($this->description())
                ->withStringParameter('sku', 'The SKU to look up')
                ->using(fn (string $sku): string => $this->execute($sku));
        }

        private function execute(string $sku): string {
            $product = \App\Domain\Products\Models\Product::query()
                ->where('sku', $sku)
                ->select(['sku', 'last_sales_count_90d', 'last_sales_count_computed_at'])
                ->first();
            if ($product === null) {
                return json_encode(['sku' => $sku, 'window_days' => 90, 'sales_count' => 0, '_note' => 'product not found'], JSON_THROW_ON_ERROR);
            }

            $payload = [
                'sku' => $sku,
                'window_days' => 90,
                'sales_count' => (int) $product->last_sales_count_90d,
            ];
            if ($product->last_sales_count_computed_at !== null) {
                $ageHours = (int) round($product->last_sales_count_computed_at->diffInMinutes(now()) / 60);
                $payload['_cache_age_hours'] = $ageHours;
                $payload['_cache_computed_at'] = $product->last_sales_count_computed_at->toIso8601String();
            } else {
                $payload['_note'] = 'sales count not yet cached';
            }

            return $this->capJson($payload, 1);
        }

        protected function reduceLargestArray(array $payload, int $maxBytes): array {
            return $payload; // small payload — never truncates in practice
        }
    }
    ```

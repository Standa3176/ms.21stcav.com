final class PromptRenderer
{
    /**
     * @return array{prompt: string, hash: string}  // hash = sha256 of rendered prompt text

    public function render(string $kind, array $context = []): array
    {
        $view = "agents.{$kind}.system";   // resolves to resources/views/agents/pricing/system.blade.php
        $rendered = view($view, $context)->render();
        return ['prompt' => $rendered, 'hash' => hash('sha256', $rendered)];
    }
}

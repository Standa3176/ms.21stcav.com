       ```php
       namespace App\Domain\Quotes\Observers;

       use App\Domain\Quotes\Exceptions\QuoteLineImmutableException;
       use App\Domain\Quotes\Models\Quote;
       use App\Domain\Quotes\Models\QuoteLine;

       final class QuoteLineImmutabilityObserver
       {
           public function saving(QuoteLine $line): void
           {
               if (! $line->exists) {
                   return; // initial creation — set ONCE per D-13
               }

               $quote = $line->quote;  // belongsTo
               if ($quote->status === Quote::STATUS_DRAFT) {
                   if ($line->isDirty('quantity_int')) {
                       $line->line_total_pence_at_quote = $line->unit_price_pence_at_quote * $line->quantity_int;
                   }
                   // Forbid price/snapshot mutation even in draft (set ONCE on creation only)
                   $forbiddenInDraft = ['unit_price_pence_at_quote', 'product_snapshot', 'sku', 'quote_id'];
                   $dirtyForbidden = array_filter(
                       array_combine($forbiddenInDraft, array_map(fn ($c) => $line->isDirty($c) ? [
                           'old' => $line->getOriginal($c), 'new' => $line->getAttribute($c),
                       ] : null, $forbiddenInDraft)),
                       fn ($v) => $v !== null
                   );
                   if (! empty($dirtyForbidden)) {
                       throw QuoteLineImmutableException::forDirtyColumns($line->id, $quote->id, $quote->status, $dirtyForbidden);
                   }
                   return;
               }

               // status != draft — ALL forbidden
               $forbidden = ['unit_price_pence_at_quote', 'line_total_pence_at_quote', 'product_snapshot', 'sku', 'quote_id', 'quantity_int'];
               $dirty = array_filter(
                   array_combine($forbidden, array_map(fn ($c) => $line->isDirty($c) ? [
                       'old' => $line->getOriginal($c), 'new' => $line->getAttribute($c),
                   ] : null, $forbidden)),
                   fn ($v) => $v !== null
               );
               if (! empty($dirty)) {
                   throw QuoteLineImmutableException::forDirtyColumns($line->id, $quote->id, $quote->status, $dirty);
               }
           }
       ```

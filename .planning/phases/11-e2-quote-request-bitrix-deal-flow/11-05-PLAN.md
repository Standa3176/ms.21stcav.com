       ```php
       namespace App\Domain\Quotes\Services;

       use App\Domain\Quotes\Models\Quote;
       use App\Domain\TradePricing\Models\CustomerGroup;
       use Illuminate\Support\Facades\DB;

       final class ImportQuoteAction
       {
           public function __construct(
               private readonly QuoteLineWriter $writer,
           ) {}

           /**
            * @param array{customer_email: string, customer_name?: string, user_id?: ?int,
            *   customer_group_id?: ?int, billing_address?: ?array,
            *   line_items: array<int, array{sku: string, quantity_int: int}>,
            *   notes?: ?string} $input
            */
           public function execute(array $input): Quote
           {
               return DB::transaction(function () use ($input) {
                   $customerGroupId = $this->resolveCustomerGroupId($input);
                   $customerGroupName = $customerGroupId
                       ? CustomerGroup::find($customerGroupId)?->name
                       : null;

                   $quote = Quote::create([
                       'user_id' => $input['user_id'] ?? null,
                       'customer_email' => $input['customer_email'],
                       'customer_name' => $input['customer_name'] ?? null,
                       'customer_group_id' => $customerGroupId,
                       'customer_group_name_at_quote' => $customerGroupName,
                       'billing_address' => $input['billing_address'] ?? null,
                       'status' => Quote::STATUS_DRAFT,
                       'expires_at' => now()->addDays(config('quote.default_expiry_days', 14)),
                   ]);

                   foreach ($input['line_items'] as $line) {
                       $this->writer->add($quote, $line['sku'], $line['quantity_int']);
                   }

                   return $quote->fresh('lines');
               });
           }

           private function resolveCustomerGroupId(array $input): ?int
           {
               if (! empty($input['user_id'])) {
                   $user = \App\Models\User::find($input['user_id']);
                   return $user?->customer_group_id;
               }
               return $input['customer_group_id'] ?? null;
           }
       }
       ```

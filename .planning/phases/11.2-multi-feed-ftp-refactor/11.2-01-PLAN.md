```php
namespace App\Domain\Competitor\Ftp\Services;

use App\Domain\Competitor\Ftp\Exceptions\FormatNormalisationException;
use App\Domain\Competitor\Services\EncodingDetector;
use ZipArchive;

/**
 * Normalises remote feed files (csv/tsv/zip/txt) to canonical CSV before they
 * land in storage/app/competitors/incoming/. Phase 5 watcher only ever sees .csv.
 *
 * D-05 (Phase 11.2 CONTEXT.md):
 *   - csv  → passthrough
 *   - tsv  → \t delimiter → , delimiter; reuse Phase 5 EncodingDetector for BOM/Windows-1252/UTF-8
 *   - zip  → ZipArchive::open → first *.csv (case-insensitive); throws if no CSV
 *   - txt  → sniff first ~100 lines for tab/comma/semicolon majority; convert to CSV; threshold 80%
 */
class FeedFormatNormaliser
{
    public function __construct(private readonly EncodingDetector $encodingDetector) {}

    /**
     * Normalise the temp file at $tempPath to canonical CSV.
     *
     * @return string Absolute path to the normalised .csv file (caller atomically moves into incoming/).
     */
    public function normalise(string $tempPath, string $format, string $localFilename): string
    {
        return match ($format) {
            'csv' => $tempPath,                                          // D-05: passthrough
            'tsv' => $this->normaliseTsv($tempPath, $localFilename),
            'zip' => $this->normaliseZip($tempPath, $localFilename),
            'txt' => $this->normaliseTxt($tempPath, $localFilename),
            default => throw FormatNormalisationException::unsupportedFormat($format),
        };
    }

    private function normaliseTsv(string $tempPath, string $localFilename): string
    {
        $encoded = $this->encodingDetector->detect($tempPath);
        $sourcePath = in_array($encoded, ['UTF-8'], true)
            ? $tempPath
            : $this->encodingDetector->convertToUtf8($tempPath, $encoded);

        $outPath = tempnam(sys_get_temp_dir(), 'feed_norm_').'.csv';
        $in = fopen($sourcePath, 'r');
        $out = fopen($outPath, 'w');
        // Strip UTF-8 BOM if EncodingDetector returned 'UTF-8-BOM' (or convertToUtf8 left one).
        $first = fgets($in);
        if ($first !== false &amp;&amp; substr($first, 0, 3) === "\xEF\xBB\xBF") {
            $first = substr($first, 3);
        }
        if ($first !== false) {
            $row = str_getcsv($first, "\t");
            fputcsv($out, $row);
        }
        while (($row = fgetcsv($in, 0, "\t")) !== false) {
            fputcsv($out, $row);
        }
        fclose($in); fclose($out);

        return $outPath;
    }

    private function normaliseZip(string $tempPath, string $localFilename): string
    {
        $zip = new ZipArchive();
        if ($zip->open($tempPath) !== true) {
            throw new FormatNormalisationException("FeedFormatNormaliser: failed to open zip {$localFilename}");
        }
        $extractTo = sys_get_temp_dir().DIRECTORY_SEPARATOR.'feed_norm_'.uniqid();
        mkdir($extractTo);

        $csvName = null;
        for ($i = 0; $i < $zip->numFiles; $i++) {
            $name = $zip->getNameIndex($i);
            if ($name === false) continue;
            if (preg_match('/\.csv$/i', $name)) { $csvName = $name; break; }
        }
        if ($csvName === null) {
            $zip->close();
            throw FormatNormalisationException::noCsvInZip($localFilename);
        }

        $zip->extractTo($extractTo, [$csvName]);
        $zip->close();
        return $extractTo.DIRECTORY_SEPARATOR.$csvName;
    }

    private function normaliseTxt(string $tempPath, string $localFilename): string
    {
        $handle = fopen($tempPath, 'r');
        $sample = '';
        $lines = 0;
        while (!feof($handle) &amp;&amp; $lines < 100) {
            $sample .= fgets($handle);
            $lines++;
        }
        fclose($handle);

        $tabs   = substr_count($sample, "\t");
        $commas = substr_count($sample, ',');
        $semis  = substr_count($sample, ';');

        $total = $tabs + $commas + $semis;
        if ($total === 0) {
            throw FormatNormalisationException::delimiterUncertain($localFilename);
        }
        $max = max($tabs, $commas, $semis);
        $confidence = $max / max($total, 1);
        if ($confidence < 0.80) {
            throw FormatNormalisationException::delimiterUncertain($localFilename);
        }

        $delim = $tabs === $max ? "\t" : ($semis === $max ? ';' : ',');

        if ($delim === ',') {
            // Already comma-separated; just rename extension by copy.
            $outPath = tempnam(sys_get_temp_dir(), 'feed_norm_').'.csv';
            copy($tempPath, $outPath);
            return $outPath;
        }

        $outPath = tempnam(sys_get_temp_dir(), 'feed_norm_').'.csv';
        $in = fopen($tempPath, 'r');
        $out = fopen($outPath, 'w');
        while (($row = fgetcsv($in, 0, $delim)) !== false) {
            fputcsv($out, $row);
        }
        fclose($in); fclose($out);
        return $outPath;
    }
}
```

```php
            $query->where('credential_id', $credId);
        }

        $feeds = $query->get();
        $this->info(sprintf('Pulling %d feed(s) (mode=%s)', $feeds->count(), $live ? 'live' : 'dry-run'));

        foreach ($feeds as $feed) {
            $this->pullOne($feed, $live);
        }

        return self::SUCCESS;
    }

    private function pullOne(CompetitorFtpFeed $feed, bool $live): void
    {
        try {
            $disk = $this->connector->connect($feed->credential);

            // D-13 step 2: skip-on-no-change
            $remoteMtime = $disk->lastModified($feed->remote_filename);
            if ($feed->remote_file_date !== null
                &amp;&amp; $remoteMtime === $feed->remote_file_date->timestamp) {
                $feed->update([
                    'last_pull_status' => 'no_change',
                    'last_pulled_at'   => now(),
                ]);
                $this->line("  no_change: {$feed->local_filename}");
                return;
            }

            if (! $live) {
                $this->line("  dry-run: would pull {$feed->remote_filename} → {$feed->local_filename}");
                return;
            }

            // D-13 step 3: download to temp
            $tempPath = tempnam(sys_get_temp_dir(), 'feed_pull_');
            $stream = $disk->readStream($feed->remote_filename);
            $out = fopen($tempPath, 'w');
            stream_copy_to_stream($stream, $out);
            fclose($out);
            if (is_resource($stream)) fclose($stream);

            // D-13 step 4: normalise
            $normalisedPath = $this->normaliser->normalise($tempPath, $feed->format, $feed->local_filename);

            // D-13 step 5: atomic move into incoming/
            $incomingPath = storage_path('app/competitors/incoming/'.$feed->local_filename);
            @mkdir(dirname($incomingPath), 0775, true);
            $tmpIncoming = $incomingPath.'.tmp';
            copy($normalisedPath, $tmpIncoming);
            rename($tmpIncoming, $incomingPath);

            // D-13 step 6: success
            $feed->update([
                'last_pulled_at'       => now(),
                'remote_file_date'     => \Illuminate\Support\Carbon::createFromTimestamp($remoteMtime),
                'last_pull_status'     => 'success',
                'last_pull_error'      => null,
                'consecutive_failures' => 0,
            ]);
            \App\Domain\Audit\Auditor::record('competitor_ftp_feed_pull', $feed->id, [
                'remote_filename' => $feed->remote_filename,
                'local_filename'  => $feed->local_filename,
            ]);

        } catch (\Throwable $e) {
            $this->handleFailure($feed, $e);
        }
    }

    private function handleFailure(CompetitorFtpFeed $feed, \Throwable $e): void
    {
        $feed->increment('consecutive_failures');
        $feed->update([
            'last_pull_status' => 'failed',
            'last_pull_error'  => substr($e->getMessage(), 0, 1000),
            'last_pulled_at'   => now(),
        ]);

        // D-10 (Phase 11.1 reused): write csv_parse_errors row
        CsvParseError::create([
            'issue_type'      => 'ftp_pull_failed',
            'filename'        => $feed->local_filename,
            'message'         => $e->getMessage(),
            'context'         => ['feed_id' => $feed->id, 'competitor_id' => $feed->competitor_id],
        ]);

        // 3-strike auto-disable + notification
        $feed->refresh();
        if ($feed->consecutive_failures >= 3 &amp;&amp; $feed->is_active) {
            $feed->update(['is_active' => false]);
            \App\Domain\Audit\Auditor::record('competitor_ftp_feed_disabled', $feed->id, [
                'reason' => 'consecutive_failures_threshold',
            ]);
            $recipients = AlertRecipient::query()
                ->where('receives_competitor_ftp_alerts', true)
                ->get();
            Notification::send($recipients, new CompetitorFtpPullFailedNotification($feed, $e->getMessage()));
        }

        $this->error("  FAILED feed_id={$feed->id} {$feed->local_filename}: ".$e->getMessage());
    }
}
```
